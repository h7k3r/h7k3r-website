                document.getElementById('user-ip').innerText = data.ip;
                document.getElementById('user-city').innerText = `${data.city.toUpperCase()}, ${data.country_code}`;
            })
            .catch(() => {
                document.getElementById('user-ip').innerText = '127.0.0.1';
                document.getElementById('user-city').innerText = 'SECURE_NODE';
            });
    </script>
